from django import forms
from .models import *
#

Formulario de la creacion de User Story

class CreateUserStoryForm(forms.ModelForm):
# 
    class Meta:
        model = UserStory
        fields = ('proyecto',
                  'nombre',
                  'descripcion',
                  'duracion_estimada',
                  'valor_negocio',
                  'prioridad',
                  'valor_tecnico',
                  'tipo_us',
                  'flujo',
                  )

        widgets = {
            'proyecto': forms.HiddenInput,
        }
