from django import forms
from .models import Usuario
from rol.models import Permiso
#

Formulario para la creacion de un Usuario

class CreateUserForm(forms.ModelForm):
#

Clase en la que se definen los datos necesarios y adicionales para inicializacion y visualizacion del formulario

    class Meta:
# 
        model = Usuario
        fields = ('username',
                  'email',
                  'first_name',
                  'last_name',
                  'estado',
                  'ci',
                  'telefono',
                  'direccion',
                  'descripcion',
                  'password',
                  'permisos'
                  )
        widgets = {
            'permisos': forms.CheckboxSelectMultiple(),
            'password': forms.PasswordInput()
        }
#

Constructor del formulario :param args: argumentos para inicializacion :param kwargs: diccionario de datos adicionales para inicializacion

    def __init__(self, *args, **kwargs):
# 
        super(CreateUserForm, self).__init__(*args, **kwargs)
        permisos_all = Permiso.objects.filter(tipo=1)
        p = self.fields['permisos'].widget
        permisos = []
        for permiso in permisos_all:
            permisos.append((permiso.id, permiso.nombre))
        p.choices = permisos
#

Metodo para guardar el formulario :param commit: indicador de guardado en la base de datos, True: se debe guardar, False: No guardar. :return: El objeto creado por el formulario luego de ejecutar el metodo en el formulario

    def save(self, commit=True):
# 
        user = super(CreateUserForm, self).save(commit=False)
        user.set_password(self.cleaned_data["password"])
        if commit:
            user.save()
            self.save_m2m()
        return user
